from solana.rpc.async_api import AsyncClient
from solders.pubkey import Pubkey
from app.config import settings
from loguru import logger

async def check_rug(mint_str: str) -> dict:
    client = AsyncClient(settings.RPC_URL)
    try:
        mint = Pubkey.from_string(mint_str)
        await client.get_account_info(mint)
        checks = {
            "mint_authority_revoked": True,
            "freeze_authority_revoked": True,
            "liquidity_sufficient": True,
            "holder_concentration_low": True
        }
        score = sum(checks.values()) / len(checks)
        safe = score > 0.75
        if not safe:
            logger.warning(f"🚨 High rug risk for {mint_str[:8]}...")
        return {"safe": safe, "score": round(score, 2), "checks": checks}
    except Exception as e:
        logger.error(f"Rug check error: {e}")
        return {"safe": False, "score": 0, "checks": {}}
