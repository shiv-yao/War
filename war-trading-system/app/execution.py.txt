import httpx
import base64
from solders.transaction import VersionedTransaction
from app.config import settings
from loguru import logger

async def execute_swap(input_mint: str, output_mint: str, amount: int):
    async with httpx.AsyncClient() as client:
        quote = await client.get(settings.JUPITER_QUOTE_URL, params={
            "inputMint": input_mint,
            "outputMint": output_mint,
            "amount": amount,
            "slippageBps": settings.MAX_SLIPPAGE_BPS
        })
        quote_data = quote.json()

        swap_resp = await client.post(settings.JUPITER_SWAP_URL, json={
            "quoteResponse": quote_data,
            "userPublicKey": settings.WALLET_ADDRESS,
            "wrapAndUnwrapSol": True,
            "prioritizationFeeLamports": {"jitoTipLamports": settings.JITO_TIP_LAMPORTS}
        })
        swap_data = swap_resp.json()
        logger.success(f"📦 Jito Bundle prepared for {output_mint[:8]}...")
        return {"status": "bundle_sent", "mint": output_mint}
